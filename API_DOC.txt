/api/join:
    TYPE: POST

    Takes the username as an argument in JSON format:
    {
        "Username": "Tripp"
    }

    upon name taken:
    returns {"Error":"name taken"}

    upon success:
    returns {"Error":"success"}

    upon general failure:
    returns {"Error":"err"}
